import re
from datetime import datetime

def parse_date(text):
    # Use a regular expression to match a day and month
    match = re.search(r'(\d{1,2})\s+(Jan|Feb|Mar|Apr|May|Jun|Jul|Aug|Sep|Oct|Nov|Dec)\s+(\d{4})', text)
    
    if match:
        # Extract the matched values from the match object
        day = int(match.group(1))
        month = match.group(2)
        year = int(match.group(3))
        
        # Create a datetime object using the extracted values and get the date in YYYY-MM-DD format
        date = datetime(year, month_to_year(month), day).strftime('%Y-%m-%d')
        return date
    
    # If no match is found, return None
    return None

def month_to_year(month):
    months = {
        'Jan': 1,
        'Feb': 2,
        'Mar': 3,
        'Apr': 4,
        'May': 5,
        'Jun': 6,
        'Jul': 7,
        'Aug': 8,
        'Sep': 9,
        'Oct': 10,
        'Nov': 11,
        'Dec': 12
    }
    
    return months[month]
